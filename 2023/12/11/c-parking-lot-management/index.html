停车场管理

2023/12/11 C/C++

题目背景

这是一道来自2327每日一题 期18的程序题

 

参考答案

原题目给出了三种做法，我个人认为后两者是类似的，因此我只选择了前两种做法发上来，法三是我做的答案，由于当时太晚了，所以用了c艹，输出格式也没有按照原题目来

法一

#include<stdio.h>
#include<string.h>
struct car{
    char state;
    int num;
    int time1,time2;
};
int main()
{
    struct car cars[100];
    int n,i,j,space=0;
    for(i=0;i<100;i++){
        cars[i].time1=0;
    }
    scanf("%d",&n);
    i=0;
    while(1){
        scanf("%c %d",&cars[i].state,&cars[i].num);
        if(cars[i].state=='E'){
            break;
        }else if(cars[i].state=='A'){
            scanf("%d",&cars[i].time1);
            if(space<n){
                space++;
                printf("car#%d in parking space #%d\n",cars[i].num,space);
            }else if(space==n){
                cars[i].state='W';
                printf("car#%d waiting\n",cars[i].num);
            }
        }else if(cars[i].state=='D'){
            scanf("%d",&cars[i].time2);
            for(j=0;j<i;j++){
              if(cars[i].num==cars[j].num){
                cars[i].time1=cars[j].time1;
        }
      }
            if(cars[i].time1==0){
                printf("the car not in park\n");
            }else{
                printf("car#%d out,parking time %d\n",cars[i].num,cars[i].time2-cars[i].time1);
                cars[i].time1=0;
                int flag=1;
                for(j=0;j<i;j++){
                    if(cars[j].state=='W'){
                        cars[j].state='A';
                        cars[j].time1=cars[i].time2;
                        flag=0;
                        printf("car#%d in parking space #%d\n",cars[j].num,space);
                        break;
                    }
                }
                if(flag){
                    space--;
                }
            }
        }
        i++;
    }
}

法二

#include<stdio.h>
#include<string.h>
 
int stack[10];
int Arrive[10];
int queue[100];
int head=0,tail=0,top=0;
int size;
void opA(int n,int t);
void opD(int n,int d);
int main()
{
    scanf("%d",&size);
    char st;
    while(scanf("%c",&st)==1)
    {
        int n,t;
 
        scanf("%d%d",&n,&t);
        switch(st)
        {
            case 'A':
                opA(n,t);
                break;
            case 'D':
                opD(n,t);
                break;
            case 'E':
                return 0;
        }
    }//while
    return 0;
}
void opA(int n,int t)
{
    if(top<size)
    {
        stack[top]=n;
        Arrive[top++]=t;
        printf("car#%d in parking space #%d\n",n,top);
    }
    else
    {
        queue[tail++]=n;
        printf("car#%d waiting\n",n);
    }
    return;
}
void opD(int n,int d)
{
    for(int i=0;i<top;i++)
    {
        if(stack[i]==n)
        {
            int t=d-Arrive[i];
            printf("car#%d out,parking time %d\n",n,t);
            for(int j=i;j<top-1;j++)
            {
                stack[j]=stack[j+1];
            }
            if(head<tail)
            {
                int x=queue[head++];
                stack[top-1]=x;
                Arrive[top-1]=d;
                printf("car#%d in parking space #%d\n",x,top);
            }
            else top-=1;
            return;
        }
    }//for
    printf("the car not in park\n");
    return;
}

法三

#include<bits/stdc++.h>
using namespace std;
typedef struct _car{
    int id,arrive;
    struct _car* next;
} car;
typedef car* pcar;
int main(){
    pcar park = new car;
    park->id = 0;
    park->next = NULL;
    int parkcnt = 0;
    pcar wait[20];
    int f=0,t=0;
    int n,maxsize;
    cin>>n;
    maxsize = n;
    while(true){
        char op;
        cin>>op;
        if(op=='E'){
            cin>>op;
            cin>>op;
            break;
        }
        else if(op=='A'){
            int id,arrive;
            cin>>id>>arrive;
            pcar tmp = new car;
            tmp->id = id;
            tmp->next = NULL;
            tmp->arrive = arrive;
            if(parkcnt<maxsize){
                pcar cur = park;
                while(cur->next!=NULL){
                    cur = cur->next;
                }
                cur->next=tmp;
                printf("car%d in\n",tmp->id);
                ++parkcnt;
            }
            else{
                printf("car%d wait\n",tmp->id);
                wait[t] = tmp;
                ++t;
            }
        }
        else{
            int id,depart;
            cin>>id>>depart;
            pcar cur=park;
            while(cur->next!=NULL){
                if(cur->next->id==id){
                    break;
                }
                cur=cur->next;
            }
            if(cur->next==NULL){
                printf("no car\n");
            }
            else{
                int time = depart-cur->next->arrive;
                printf("car%d out,time=%d\n",id,time);
                pcar tmp = cur->next;
                cur->next=cur->next->next;
                delete tmp;
                --parkcnt;
            }
            if(f<t){
                pcar tmp = wait[f];
                pcar cur = park;
                while(cur->next!=NULL){
                    cur = cur->next;
                }
                cur->next=tmp;
                tmp->arrive=depart;
                printf("car%d in\n",tmp->id);
                ++parkcnt;
                ++f;
            }
        }
    }
}

 

算法分析

法一
用了一个大数组cars储存所有的车辆，同时每个车辆额外分配了空间储存车辆的状态A/W（分别对应arrived和waiting），通过维护状态来进行车辆的进出。
当接到A命令的时候查询余量是否足够，判断新进去的车辆的状态置A/W。当接到D命令的时候寻找对应的车辆是否存在，再寻找有没有状态为W的车辆可以进入。

法二
属于是比较常规的数据结构的思路了。停车场的机制很像一个栈，只能从一端进出，而等待的车辆的机制很像一个队列，先到先出，因此利用了stack和queue实现。
估计是官方答案，也就是题目原作的解法，是可以AC的。
但是用栈实现有一个问题，在D指令的时候，不可避免地要进行r次查询和n-r次移动，因此效率比较低。

法三
是我的做法，语言上选用了C艹，主要是因为当时写题目的时候已经凌晨三点钟了，所以想着快点解决（应该用py的），就用了C艹，不过没有涉及C艹的任何特性，只使用了cin，cout，new，delete四个东西，且都可以很快的改写为C，所以我就不改了（硬气）。
我用了链表代替栈，因为离开的时候并不改变车辆之间的相互关系，而查询的时候不可避免地要进行一个遍历（当然也有一种做法就是另开一个储存位置的数组idx[]，这种空间换时间的方法就见仁见智了），因此便于增删的链表是更合适的。
除了链表以外实现方法与法二基本类似，由于是现写的，码风有些暴力，不过问题不大哈哈哈哈哈哈
