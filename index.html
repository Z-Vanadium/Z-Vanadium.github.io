#define ElementType int

void swap(ElementType* x, ElementType* y){
    ElementType tmp;
    tmp = *x;
    *x = *y;
    *y = tmp;
}

void BubbleSort(ElementType array[], int n){
    int p,i;
    for(p=1;p<n;++p){
        for(i=0;i<n-p;++i){
            if(array[i]>array[i+1]){
                swap(array+i,array+i+1);
            }
        }
    }
}

插入排序

#define ElementType int

void InsertionSort(ElementType array[], int n){
    int p,i;
    ElementType tmp;
    for(p=1;p<n;++p){
        tmp = array[p];
        i=p;
        while(i>0 && array[i-1]>tmp){
            array[i] = array[i-1];
            --i;
        }
        array[i] = tmp;
    }
}

void ShellSort(ElementType array[], int n){
    int Sedgewick[] = {929,505,209,109,41,19,5,1,0};
    int s,d;
    s=0;
    while(Sedgewick[s]>n)   ++s;
    int p,i;
    ElementType tmp;
    for(d=Sedgewick[s];d>0;d=Sedgewick[++s])
        for(p=d;p<n;++p){
            tmp = array[p];
            i=p;
            while(i>=d && array[i-d]>tmp){
                array[i] = array[i-d];
                i-=d;
            }
            array[i] = tmp;
        }
}

归并排序 | 最喜欢的排序，小时候写这个的时候全村抗洪

#define ElementType int

void merge(ElementType array[],ElementType assist[],int start,int end){
    int i,j,k,mid;
    mid = (start+end)/2;
    i = k = start;
    j = mid+1;
    while(i<=mid && j<=end){
        if(array[i]<array[j])   assist[k++] = array[i++];
        else    assist[k++] = array[j++];
    }
    while(i<=mid)   assist[k++] = array[i++];
    while(j<=end)   assist[k++] = array[j++];

    for(i=0;i<=end;++i) array[i] = assist[i];
}

void MergeSort_(ElementType array[], ElementType assist[], int start, int end){
    int mid;
    if(start<end){
        mid = (start+end)/2;
        MergeSort_(array,assist,start,mid);
        MergeSort_(array,assist,mid+1,end);
        merge(array,assist,start,end);
    }
}

void MergeSort(ElementType array[], int n){
    ElementType* assist = malloc(sizeof(ElementType)*n);
    MergeSort_(array,assist,0,n-1);
    free(assist);
}

void Qsort(ElementType array[],int left,int right){
    if(left>=right) return;
    int low,high,key;
    low = left;
    high = right;
    key = left;
    while(low<high){
        while(low<high && array[key]<=array[high])  --high;
        while(low<high && array[low]<=array[key])   ++low;
        swap(array+low,array+high);
    }
    swap(array+key,array+high);
    key = high;
    Qsort(array,left,key-1);
    Qsort(array,key+1,right);
}

void QuickSort(ElementType array[], int N){
    Qsort(array,0,N-1);
}

桶排序 | 范围内的王者

#include<stdlib.h>
#define LOW 1
#define HIGH 10

void BucketSort(int array[],int N){
    int* cnts = (int*)malloc(sizeof(int)*(HIGH-LOW+1));
    int len = HIGH - LOW + 1;
    int i;
    for(i=0;i<len;++i)  cnts[i] = 0;
    for(i=0;i<N;++i)    ++cnts[array[i]];
    int cur;
    cur=0;
    for(i=0;i<len;++i){
        int tmp = cnts[i];
        while ((tmp>0)){
            array[cur] = i;
            --tmp;
            ++cur;
        }
    }
}

基数排序 | 桶排序升级版，那么代价是什么

#include<stdio.h>
#include<stdlib.h>
#define MAXDIGIT 3
#define RADIX 10

typedef struct Node *PtrToNode;
struct Node{
    int key;
    PtrToNode next;
};
struct HeadNode{
    PtrToNode head,tail;
};
typedef struct HeadNode Bucket[RADIX];

int GetDigit(int val,int digit){
    int ret;
    while(digit){
        --digit;
        ret = val%RADIX;
        val /= RADIX;
    }
    return ret;
}

void MSDSort(int array[],int L, int R, int D){
    int Di,i,j;
    Bucket B;
    PtrToNode tmp,p,List = NULL;
    if(D==0)    return;
    for(i=0;i<RADIX;++i)    B[i].head = B[i].tail = NULL;
    for(i=L;i<=R;++i){
        tmp = (PtrToNode)malloc(sizeof(struct Node));
        tmp->key = array[i];
        tmp->next = List;
        List = tmp;
    }
    p = List;
    while(p){
        Di = GetDigit(p->key,D);
        tmp = p;
        p = p->next;
        if(B[Di].head == NULL)  B[Di].tail = tmp;
        tmp->next = B[Di].head;
        B[Di].head = tmp;
    }
    i = j = L;
    for(Di=0;Di<RADIX;++Di){
        if(B[Di].head){
            p = B[Di].head;
            while(p){
                tmp = p;
                p = p->next;
                array[j++] = tmp->key;
                free(tmp);
            }
            MSDSort(array,i,j-1,D-1);
            i = j;
            
        }
    }
}

void RadixSort(int array[], int N){
    MSDSort(array,0,N-1,MAXDIGIT);
}

最小生成树 | Prim算法与 Kruskal算法
2023/12/7

0. 什么是最小生成树

最小生成树（Minimum Spanning Tree）这个名词乍一听有点不知所云，但如果我们分拆每一个词，很快就能理解其含义。

  • 树，顾名思义，这是一棵树。我们知道，树是一类特殊的图，特殊树不存在环，一定是连通的，并且一棵有 N 个节点的树有且仅有 N-1 条边。

  • 生成，这是一个专业术语，意思是这棵树的节点和边都是来自某一个图的，并且树的节点数等于图的节点数。

  • 最小，意思是路径最短，或者根据题目要求，某项或某几项最小。

因此，最小生成树问题即可翻译为从图中选出 N-1 条边使 N 个节点构成一棵树，并且使这棵树满足某一个特定权重最小的条件。

1. 两种算法 | 归与并

根据 MST 的定义，我们可以构思出两种解决这个问题的方法，分别是归和并两种方法。（这个名字是作者自己瞎起的）

按照我们所学的知识，和树有关的概念有哪些？节点，和森林。

因此，如果我们想要生成一棵最小的树，我们可以考虑由节点生成，和由森林退化两个角度——也就是归与并。

当然，我们想到这两点是远不足以获得Turing奖的，因为Prim与Kruskal两个人老早就提出了这两种算法，同时还以自己的名字命好了名。

2. 小树快快长 | Prim算法

Prim算法是节点生成树。

首先，从图中任选一个节点S作为MST的根节点。然后，每一次循环都分别从MST的每一个节点出发，寻找一条满足路径最短且不会使MST构成环的边，加入到MST中。当已经寻找到 N-1 条边或找不到满足上述这样条件的边时，循环结束。

通俗一点就是，Prim算法是先有了一棵MST，然后通过生长的方式把其他节点依次加入进去。

伪代码描述如下：

Prim(S){
    new MST;
    MST.putin(S);
    while(True){
        V = 路径最短且不会使MST构成环;
        E = 路径最短且不会使MST构成环的边;
        if( MST.size == N-1 or V == NULL){
            break;
        }
        MST.putin(V,E);
    }
    return MST;
}

可以看到，伪代码的叙述是十分简洁易懂的（这是作者基于C/Python/Java发明的新语言！），但是我们在实现的时候需要考虑一些至关重要的问题。

  • Q：我们一定要实现一个MST吗？

  • A：当然不需要，在绝大多数问题中，我们只需要一个parent数组来表示某节点的父亲就可以了（同时我们需要一个集合判断某个节点是否属于MST）

  • Q：如何实现寻找V和E的操作？

  • A：首先是路径最短，我们可以直接遍历图的每一条边，也可以以最小堆的形式储存图，也可以先对边按权重为关键字排序，这个任君选择；其次是不会使MST构成环，这个也简单，只要该边的终点节点不属于MST即可。

  • Q：一定有MST吗？

  • A：如果题目没有担保，那就不一定，最后可以根据MST.size()的值判断是否得到的是MST。

3. 森林会记住一切 | Kruskal算法

Kruskal算法是森林生成树。

首先，算法将图中所有节点视作一棵树（孤零零的也很可爱捏），这样算法就可以简化为考虑树与树连接成MST了。然后，寻找权重最小，且不会产生连通的边加入到MST中，当已经寻找到 N-1 条边或找不到满足上述这样条件的边时，循环结束。

相对于Prim算法，Kruskal算法更加容易理解。就是用能用的最短边来连接图中的两棵树，最后把所有已有的树连接成一棵MST。

依然是伪代码，描述如下：

Kruskal(){
    new MST;
    while(True){
        E = 路径最短且不会使MST构成环的边;
        V, W = E的出发节点的终点节点;
        if( MST.size == N-1 or V == NULL){
            break;
        }
        MST.putin(E)
    }
    return MST;
}

可以看到，Kruskal算法与Prim算法是很类似的，当然同样有一些关键问题要回答。

  • Q1：如何寻找权重最短的边？

  • A1：同上，我们也有三种办法，分别是直接遍历，最小堆，先对边按权重为关键字排序。

  • Q2：如何判断是否会构成环？

  • A2：有且仅有一种情况会构成环，那就是V和W都是某棵树的节点，因此我们只需要判断V和W是不是属于同一棵树即可，实现时我们可以定义root数组，储存每一个节点的根节点，当然，初始化为自己，表示一棵只有一个根节点的树。

4. 算法本质 | 九九归一

不管是Prim算法还是Kruskal算法，其本质都是贪心算法。可以证明，每一步贪心最后可以得到最优解。

数据结构杂谈（二）| 如何学习数据结构
2023/12/6

数据结构这一块已经零零碎碎学了一个月了。

学下来唯一的感觉就是：快。

这种快一半是教学进度之快，一半是学习速度之快。

在中国大学MOOC平台上，陈越、何钦明老师的数据结构课程总共分为 12 周，目前进行至第 8 周，但是已经讲到了最小生成树和拓扑排序（虽然这些知识对于洪神来说小菜一碟） ^洪神 。

简单梳理一下目前为止都讲了些什么吧。

第 1 周，数据结构的概念和算法的概念。

第 2 周，线性结构、堆栈和队列。由于学过C程，数据结构略去了数组和链表，直接介绍了堆栈和队列，但是目前为止这些东西对于高中选课包含技术的同学来说都还是很简单的。

第 3-5 周，树。按照树、二叉树及其遍历、BST、AVL树、堆、Huffman树，最后是并查集的思路展开。

第 6-8 周，图。按照图、图的表示、BFS和DFS、最短路径、最小生成树和拓扑排序展开。这些内容就涉及到了一些经典算法的学习了。

然而经典算法之所以是经典算法，其原因并不仅仅在于其优秀的问题解决能力，更在于后人一次次地修补优化，这也就导致很多同学（包括我）学习这些算法时，经常觉得“这也太超出常理了吧，怎么想的？”，但是当你仔细研读之后，你会稍稍理解，但还是不忍直呼“妙！”。

不过，可惜的是，不论你怎么怎么和经典算法产生共鸣，当你上机尝试时，90% 的概率一行代码都写不出来，99% 的概率程序就是跑不出正确结果，一直WAWAWAWA。

这种情况尤其频繁地发生在图这一章。

由于图的抽象性和复杂性，涉及图的算法常常也是抽象且复杂的。例如最短路径中的 Dijstra 算法，初学时你可能很难很难理解基点节点的作用和缘由，以及好奇这个算法为什么就能得到单源最短路径。

【夜深人静码代码】旅游规划 | Dijkstra算法
2023/12/6

原题目：

可以看到，这是一道图的题目，要求我们找到节点 S 到 D 的最短路径，存在多条最短路径时，输出收费最低的。

思路：

不难想到，这是一道双条件最短路径问题，解决此类问题，我们常用Dijkstra算法或Floyd算法，这里由于是单源对单源，我们选用时间复杂度上更优的 Dijkstra 算法。

    scanf("%d %d %d %d",&N,&M,&S,&D);
    int i,j,V;
    for(i=0;i<N;++i)
        for(j=0;j<N;++j){
            dist[i][j] = -1;
            cost[i][j] = -1;
        }
    for(i=0;i<M;++i){
        int x,y,d,c;
        scanf("%d %d %d %d",&x,&y,&d,&c);
        dist[x][y] = d;
        dist[y][x] = d;
        cost[x][y] = c;
        cost[y][x] = c;
    }

程序输入部分，我们采用邻接矩阵储存图，除了分别用 dist 和 cost 两个邻接矩阵储存图，我们也可以定义结构体矩阵来储存，对于这道题，两者并无优劣。

由于我们选用了 Dijkstra 算法，我们需要：

再定义一个数组 distance[V] ，储存源点节点 S 到 节点V 的最短路径（初始化为 INF）；

一个数组 money[V] ，储存源点节点 S 到节点 V 在保证路径最短的情况下的最少花费（初始化为 INF）；

以及一个数组 visited[V] ，表示节点 V 是否已经确定为路径最短的节点。

    for(i=0;i<N;++i){
        distance[i] = INF;
        money[i] = INF;
        visited[i] = 0;
    }

做完这么多准备工作，我们可以开始书写 Dijkstra 算法了。

Dijkstra 算法可以描述为：

对图进行 N-1 次遍历，其中每一次都会提前确定一个基点节点 baseV（最开始是源点 S），并对所有节点进行遍历，对于其中每个节点 V 做如下判断：若该节点 V 未被确定为路径最短节点，且从源点 S 到基点节点 baseV 的最短距离加上基点节点 baseV 到该节点 V 的距离小于 源点 S 到该节点 V 的最短距离，就更新之；然后再次遍历所有节点，选取未被确定为路径最短，并且目前距离源点 S 最近的的节点作为新的 baseV 下一次循环。

这个算法看起来很抽象，实则可以理解为 BFS 和贪心的结合体。

每次遍历进行一次 BFS，贪心选取距离源点最近的节点作为基点，再借助这个基点节点更新其他节点的最短距离（术语称之为松弛），循环往复 N-1 次即可获得答案。

剩余代码如下：

int baseV;
    baseV = S;

    visited[S] = 1;
    distance[S] = 0;
    money[S] = 0;
    
    for(i=0;i<N-1;++i){
        for(V=0;V<N;++V){
            if(visited[V] == 0 && dist[baseV][V]>0 && (distance[baseV] + dist[baseV][V] < distance[V])){
                distance[V] = distance[baseV] + dist[baseV][V];
                money[V] = money[baseV] + cost[baseV][V];
            }
        }

        int min;
        min = INF;

        for(V=0;V<N;++V){
            if(visited[V] == 0 && distance[V]  < min ){
                min = distance[V];
                baseV = V;
            }
        }
        visited[baseV] = 1;
    }        

    
    printf("%d %d",distance[D],money[D]);

但是这个程序连样例点都过不了，为什么？

因为题目要求我们，在有多条最短路径时，找到花费最少的，换言之就是在从源点 S 到基点节点 baseV 的最短距离加上基点节点 baseV 到该节点 V 的距离等于 源点 S 到该节点 V 的最短距离时，根据花费的变化决定是否进行更新。

改进如下：

    int baseV;
    baseV = S;

    visited[S] = 1;
    distance[S] = 0;
    money[S] = 0;
    
    for(i=0;i<N-1;++i){
        for(V=0;V<N;++V){
            if(visited[V] == 0 && dist[baseV][V]>0 && (distance[baseV] + dist[baseV][V] < distance[V])
            ||(distance[baseV] + dist[baseV][V] == distance[V] && money[V] > money[baseV] + cost[baseV][V])){
                distance[V] = distance[baseV] + dist[baseV][V];
                money[V] = money[baseV] + cost[baseV][V];
            }
        }

        int min;
        min = INF;

        for(V=0;V<N;++V){
            if(visited[V] == 0 && distance[V]  < min ){
                min = distance[V];
                baseV = V;
            }
        }
        visited[baseV] = 1;
    }        
    
    printf("%d %d",distance[D],money[D]);

具体思路就是利用堆储存邻接矩阵，这样可以降低寻找离源点最近的节点时的复杂度，C++自带的 stl 提供了堆 priority_queue，同时在「手写」二叉堆的情况下，时间复杂度可以降低到 O(NlogN)。（不过应该没人会手写二叉堆解决这个问题吧）

堆：堆操作的本质是维护（二）
2023/12/6

0. 择优而取 | 堆的建立

在上一篇的结尾，我们留下了一个问题，就是如何建立一个堆。

在建立一个堆之前，我们需要数据。假设我们已经有了N个数据，储存在数组Array里面，接下来我们的任务就是利用Array的数据建立一个堆Heap。

回忆上一篇文章，我们主要讨论了堆的插入与删除。而为了实现这两个操作，我们专门分别设计了两个维护函数，分别是向上和向下维护的函数，而我们建立一个堆的过程，恰好就是一个破坏-维护的过程。

为什么这么说，首先假定我们有了一个拥有N-1个节点/元素的堆Heap，我们要插入一个数据，获得一个N个节点的堆，那我们插入的新数据，不正是破坏么，而我们接下来要做的就是维护这个被破坏的堆。最后重复这样的操作N次，我们就可以得到一个合法的大堆。

接下来我们要书写代码了么？

先等等，我们要知道，维护一个堆是有两种方法的，分别向上和向下维护，那么显而易见的是，建立一个堆应该也有两种方法。

首先考虑向上维护来建立一个堆。

如果要向上维护，我们要做的很简单，对于Array里面的每一个数，我们只需要将其插入Heap当中就可以了。

回顾一下原先的Insert函数代码。

int Insert(MaxHeap MaxH, ElementType x){
    if(IsFull(MaxH))    return ERROR;
    MaxH->MaxSize+=1;

    int child=MaxH->MaxSize;
    ElementType* array=MaxH->Array;

    array[child]=x; //x入堆
    return UpShift(MaxH, child);
}

我们唯一要做的事情就是遍历数组Array，对其中所有元素都进行一次插入操作，这个堆就建立好了。

我们考虑其时间复杂度，对于第 i 个数据，我们要进行比较交换的次数为 logi 次，因此时间复杂度为 O(nlogn) ，这个时间复杂度已经很优秀了。

接下来考虑向下维护。

如果我们要向下维护一个堆，我们一定要保证这个堆的子堆是合法的堆。如果我们一个数据一个数据插入，我们能保证子堆，也就是最底下的堆是合法的堆吗？

显然不行。因为我们插入到堆底的数据是任意的，不一定就小于其父堆。

因此我们考虑从最后一个非叶节点开始，向上维护堆。

下面给出代码。

MaxHeap ArrayToMaxHeap(ElementType* array, int N){  //通过array建立MaxHeap，返回建立的MaxHeap
    MaxHeap MaxH = BuildMaxHeap();
    MaxH->MaxSize = N;
    int i;
    for(i=0;i<N;++i)    MaxH->Array[i+1]=array[i];
    for(i=N/2;i>=1;--i)    DownShift(MaxH, i);
    return MaxH;
}

首先我们建立一个空堆MaxH，大小比Array大1（注意到MaxSize指向的是MaxH下标的最后一个，而不是MaxH的大小）。

接下来我们无视堆的规则，直接将Array存入堆，这样一个堆是一个无序的堆。

然后我们从堆的 N/2 处开始向下维护，（不从N处开始是为了剪枝，因为N/2+1 ~ N处的节点都没有子堆），如此维护到根节点，一个堆就建立好了。

考虑其时间复杂度，可以发现对于数组Array，我们进行了 1 次遍历和 N/2 次向下维护，而向上维护的复杂度为 O(logn) ，根据数学知识，我们可以计算得出向下维护建堆的复杂度为 O(N) 。

为什么是logn而不是nlogn？参见这篇文章（需要一些数学知识，因此这里Vanadium略去不写）
可以看到，向下维护建堆的方法优于向上建堆（插入建堆），因此我们择优而取，选择向下维护的方法建堆。

1. 优秀的原地排序算法 | 堆排序

所以……有了一个堆，我们能做什么呢？

接下来，我们就可以做很多事情了！

比方说，堆排序。

排序算法大家应该不陌生，在C程课上，我们学习过冒泡排序、选择排序、交换排序等一系列 O(N2) 排序算法，也有接触过高效的快速排序等排序算法。

然而在我们有了堆的概念之后，我们是否可以做一个全新的排序算法呢？

我们知道，堆的特点就是根节点的值永远是最大（小）值，所以我们只需要重复删除堆的节点，并把删除的节点依次输出即可。

所以我们的问题就是，如何建立一个堆，巧的是，这个问题刚才我们已经解决过了，就是向下维护建堆。

我们可以轻松的给出代码。

void HeapSort(ElementType* array, int N){   //对array原地进行堆排序
    MaxHeap MaxH = ArrayToMaxHeap(array, N);
    int i;
    for(i=0;i<N;++i)   array[i]=Delete(MaxH);
    return;
}

可以看到，我们在解决了建堆、删除两个操作之后，实现堆排序的算法不要太简单。

不过有细心的朋友可能会注意到Vanadium所说的“原地排序”，但是代码实现里面根本不是原地排序，因为额外申请了一个Heap的空间。

实际上这是因为，我们的代码是实例化一个ADT，是需要封装的，如果我们只是应用堆的思想，而不拘泥于堆的格式，我们可以直接利用Array进行排序——毕竟Array和Heap的array明显是一样的。

具体的代码实现就不书写了。

HeapSort的时间复杂度也是 O(nlong)，而空间复杂度为 O(1) ，并且效率基本稳定。因此在一些容易MLE（虽然很少）的大量数据的题中，HeapSort会体现出它的优势。
